154 /*
155  * The result class is responsible for producing a printable string representation of a benchmark run.
156  */
157 class result {
158  size_t d_num_iterations;
159  std::chrono::nanoseconds d_duration;
160  size_t d_num_bytes;
161 
162 public:
163  result(size_t num_iterations, std::chrono::nanoseconds duration, size_t num_bytes)
164  : d_num_iterations(num_iterations)
165  , d_duration(duration)
166  , d_num_bytes(num_bytes)
167  {}
168 
169  size_t get_ns_per_op() const {
170  if (d_num_iterations <= 0) {
171  return 0;
172  }
173  return d_duration.count() / d_num_iterations;
174  }
175 
176  double get_mb_per_s() const {
177  if (d_num_iterations <= 0 || d_duration.count() <= 0 || d_num_bytes <= 0) {
178  return 0;
179  }
180  return ((double(d_num_bytes) * double(d_num_iterations) / double(1e6)) /
181  double(std::chrono::duration_cast<std::chrono::seconds>(d_duration).count()));
182  }
183 
184  std::string to_string() const {
185  std::stringstream tmp;
186  tmp << std::setw(12) << std::right << d_num_iterations;
187  size_t npo = get_ns_per_op();
188  tmp << std::setw(12) << std::right << npo << std::setw(0) << " ns/op";
189  double mbs = get_mb_per_s();
190  if (mbs > 0.0) {
191  tmp << std::setw(12) << std::right << mbs << std::setw(0) << " MB/s";
192  }
193  return std::string(tmp.str());
194  }
195 };

225 public:
226  context(const benchmark_info& info, const options& opts)
227  : d_timer_on(false)
228  , d_start()
229  , d_duration()
230  , d_benchtime(std::chrono::seconds(opts.get_benchtime()))

237  size_t num_iterations() const { return d_num_iterations; }

242  void start_timer() {
243  if (!d_timer_on) {
244  d_start = std::chrono::high_resolution_clock::now();
245  d_timer_on = true;
246  }
247  }
248  void stop_timer() {
249  if (d_timer_on) {
250  d_duration += std::chrono::high_resolution_clock::now() - d_start;
251  d_timer_on = false;
252  }
253  }
254  void reset_timer() {
255  if (d_timer_on) {
256  d_start = std::chrono::high_resolution_clock::now();
257  }
258  d_duration = std::chrono::nanoseconds::zero();
259  }

261  void set_bytes(int64_t bytes) { d_num_bytes = bytes; }

263  size_t get_ns_per_op() {
264  if (d_num_iterations <= 0) {
265  return 0;
266  }
267  return d_duration.count() / d_num_iterations;
268  }

270  void run_n(size_t n) {
271  d_num_iterations = n;
272  reset_timer();
273  start_timer();

278  void run_parallel(std::function<void(parallel_context*)> f) {
279  parallel_context pc(d_num_iterations);
280  std::vector<std::thread> threads;
281  for (size_t i = 0; i < d_num_threads; ++i) {
282  threads.push_back(std::thread([&pc,&f]() -> void {
283  f(&pc);
284  }));
285  }
286  for(auto& thread : threads){
287  thread.join();
288  }
289  }
290 
291  result run() {
292  size_t n = 1;
293  run_n(n);
294  while (d_duration < d_benchtime && n < 1e9) {
295  size_t last = n;
296  if (get_ns_per_op() == 0) {
297  n = 1e9;
298  } else {
299  n = d_duration.count() / get_ns_per_op();
300  }
301  n = std::max(std::min(n+n/2, 100*last), last+1);
302  n = round_up(n);
303  run_n(n);
304  }
305  return result(n, d_duration, d_num_bytes);
306  }
307 
308 private:
309  template<typename T>
310  T round_down_10(T n) {
311  int tens = 0;
312  while (n > 10) {
313  n /= 10;
314  tens++;
315  }
316  int result = 1;
317  for (int i = 0; i < tens; ++i) {
318  result *= 10;
319  }
320  return result;
321  }
322 
323  template<typename T>
324  T round_up(T n) {
325  T base = round_down_10(n);
326  if (n < (2 * base)) {
327  return 2 * base;
328  }
329  if (n < (5 * base)) {
330  return 5 * base;
331  }
332  return 10 * base;
333  }

336 /*
337  * The run_benchmarks function will run the registered benchmarks.
338  */
339 void run_benchmarks(const options& opts) {
340  std::regex match_r(opts.get_bench());
341  auto benchmarks = registration::get_ptr()->get_benchmarks();
342  for (auto& info : benchmarks) {
343  if (std::regex_match(info.get_name(), match_r)) {
344  context c(info, opts);
345  auto r = c.run();
346  std::cout << std::setw(35) << std::left << info.get_name() << r.to_string() << std::endl;
347  }
348  }
349 }

362  benchpress::options bench_opts;
363  try {
364  cxxopts::Options cmd_opts(argv[0], " - command line options");
365  cmd_opts.add_options()
366  ("bench", "run benchmarks matching the regular expression", cxxopts::value<std::string>()
367  ->default_value(".*"))
368  ("benchtime", "run enough iterations of each benchmark to take t seconds", cxxopts::value<size_t>()
369  ->default_value("1"))
370  ("cpu", "specify the number of threads to use for parallel benchmarks", cxxopts::value<size_t>()
371  ->default_value(std::to_string(std::thread::hardware_concurrency())))
372  ("help", "print help")
373  ;
374  cmd_opts.parse(argc, argv);
375  if (cmd_opts.count("help")) {
376  std::cout << cmd_opts.help({""}) << std::endl;
377  exit(0);
378  }
379  if (cmd_opts.count("bench")) {
380  bench_opts.bench(cmd_opts["bench"].as<std::string>());
381  }
382  if (cmd_opts.count("benchtime")) {
383  bench_opts.benchtime(cmd_opts["benchtime"].as<size_t>());
384  }
385  if (cmd_opts.count("cpu")) {
386  bench_opts.cpu(cmd_opts["cpu"].as<size_t>());
387  }
388  } catch (const cxxopts::OptionException& e) {
389  std::cout << "error parsing options: " << e.what() << std::endl;
390  exit(1);
391  }
392  benchpress::run_benchmarks(bench_opts);
393  float duration = std::chrono::duration_cast<std::chrono::milliseconds>(
